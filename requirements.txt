openai==1.14.0

# ==========================================

# ==========================================

fastapi==0.110.0

# ==========================================
# BASE DE DATOS Y ORM
# ==========================================
# pymysql: El driver (conector) puro de Python que permite hablarle a bases de datos MySQL (síncrono).
pymysql==1.1.0
# aiomysql: El driver ASÍNCRONO para MySQL.
# Elegido obligatoriamente para producción con FastAPI. Evita que el servidor Python se "congele"

aiomysql==0.2.0
# cryptography: Requerido por algunos sistemas modernos para cifrar conexiones MySQL de forma segura.
cryptography==42.0.5
# sqlalchemy: El ORM (Object-Relational Mapper). 
# Elegido para poder programar las bases de datos (Users, Messages) como "Clases" de Python (models.py)

sqlalchemy==2.0.28

# ==========================================

pydantic==2.6.4

# ==========================================

# ==========================================

requests==2.31.0
httpx==0.27.0

# ==========================================

# ==========================================

# Elegida ESPECÍFICAMENTE porque la base de datos de PHP tiene las descripciones de los

chromadb==0.4.24
# sentence-transformers: Librería matemática de HuggingFace.
